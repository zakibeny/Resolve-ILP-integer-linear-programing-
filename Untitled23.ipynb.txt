        "# -*- coding: utf-8 -*-\n",
        "\"\"\"\n",
        "تطبيق Streamlit لخوارزمية AHRH مع الحفاظ على التوازي (ThreadPoolExecutor)\n",
        "يمكنك:\n",
        "- توليد مسائل عشوائية بأبعاد تختارها\n",
        "- رفع ملفات مسائل حقيقية (بصيغة Körkel-Ghosh)\n",
        "- تعديل معاملات الخوارزمية (عدد الدورات، حجم المجموعة الخشنة، الصبر)\n",
        "- رؤية النتائج وتطور الفجوة بشكل تفاعلي\n",
        "\"\"\"\n",
        "\n",
        "import streamlit as st\n",
        "import numpy as np\n",
        "import pulp\n",
        "import time\n",
        "import matplotlib.pyplot as plt\n",
        "import pandas as pd\n",
        "from concurrent.futures import ThreadPoolExecutor, as_completed\n",
        "import warnings\n",
        "warnings.filterwarnings(\"ignore\")\n",
        "\n",
        "# ------------------- إعدادات التوازي -------------------\n",
        "NUM_WORKERS = 4  # يمكن تعديله حسب الحاجة\n",
        "\n",
        "# ===================== دوال الخوارزمية الأساسية =====================\n",
        "\n",
        "def solve_lp_fixed_y_uflp(y_int, f, c):\n",
        "    \"\"\"حل مشكلة التخصيص لـ UFLP بمتجه y محدد (المرافق المفتوحة)\"\"\"\n",
        "    open_fac = np.where(y_int > 0.5)[0]\n",
        "    if len(open_fac) == 0:\n",
        "        return float('inf')\n",
        "    total = np.sum(f[open_fac])\n",
        "    for j in range(c.shape[1]):\n",
        "        total += min(c[i, j] for i in open_fac)\n",
        "    return total\n",
        "\n",
        "def lp_relaxation_uflp(f, c):\n",
        "    n, m = len(f), c.shape[1]\n",
        "    lp = pulp.LpProblem(\"UFLP_LP_Relax\", pulp.LpMinimize)\n",
        "    y = [pulp.LpVariable(f\"y_{i}\", 0, 1) for i in range(n)]\n",
        "    x = {(i, j): pulp.LpVariable(f\"x_{i}_{j}\", 0) for i in range(n) for j in range(m)}\n",
        "    lp += pulp.lpSum(f[i] * y[i] for i in range(n)) + pulp.lpSum(c[i, j] * x[(i, j)] for i in range(n) for j in range(m))\n",
        "    for j in range(m):\n",
        "        lp += pulp.lpSum(x[(i, j)] for i in range(n)) == 1\n",
        "    for i in range(n):\n",
        "        for j in range(m):\n",
        "            lp += x[(i, j)] <= y[i]\n",
        "    solver = pulp.PULP_CBC_CMD(msg=False)\n",
        "    lp.solve(solver)\n",
        "    if lp.status == 1:\n",
        "        y_val = np.array([pulp.value(y[i]) for i in range(n)])\n",
        "        obj = pulp.value(lp.objective)\n",
        "        return y_val, obj\n",
        "    return None, None\n",
        "\n",
        "def get_fractional_indices(y, eps=0.01):\n",
        "    return np.where((y > eps) & (y < 1 - eps))[0]\n",
        "\n",
        "def compute_R(y):\n",
        "    return np.max(np.minimum(np.abs(y), np.abs(1 - y)))\n",
        "\n",
        "def generate_biased_directions(y_lp, frac_idx, count, alpha, bias_strength=0.5):\n",
        "    n_free = len(frac_idx)\n",
        "    if y_lp is None or n_free == 0:\n",
        "        dirs = np.random.randn(count, max(1, n_free))\n",
        "        dirs = dirs / (np.linalg.norm(dirs, axis=1, keepdims=True) + 1e-12)\n",
        "        return dirs\n",
        "    y_frac_target = np.clip(y_lp[frac_idx], 0, 1)\n",
        "    base_dir = y_frac_target - 0.5\n",
        "    if np.linalg.norm(base_dir) > 0:\n",
        "        base_dir = base_dir / np.linalg.norm(base_dir)\n",
        "    else:\n",
        "        base_dir = np.zeros(n_free)\n",
        "    dirs = []\n",
        "    for _ in range(count):\n",
        "        rand = np.random.randn(n_free)\n",
        "        rand = rand / (np.linalg.norm(rand) + 1e-12)\n",
        "        u = bias_strength * base_dir + (1 - bias_strength) * rand\n",
        "        u = u / (np.linalg.norm(u) + 1e-12)\n",
        "        dirs.append(u)\n",
        "    return np.array(dirs)\n",
        "\n",
        "def hierarchical_radial_scan_parallel(y_center, R, n_free, f, c, best_cost, best_y,\n",
        "                                      n_layers=2, dirs_per_layer=10, alpha_schedule='adaptive',\n",
        "                                      y_lp=None, gap_threshold=5.0):\n",
        "    frac_idx = get_fractional_indices(y_center)\n",
        "    if len(frac_idx) == 0:\n",
        "        return best_cost, best_y\n",
        "    y_frac = y_center[frac_idx].copy()\n",
        "    n_free_actual = len(frac_idx)\n",
        "    local_best = best_cost\n",
        "    local_best_y = best_y\n",
        "    base_alpha = R / (np.sqrt(n_free) + 1e-12) if n_free > 0 else 0.1\n",
        "\n",
        "    for layer in range(1, n_layers + 1):\n",
        "        if alpha_schedule == 'adaptive':\n",
        "            alpha_k = base_alpha * np.exp(- (layer - 1) / n_layers)\n",
        "        else:\n",
        "            alpha_k = (layer / n_layers) * base_alpha\n",
        "        dirs_count = max(3, dirs_per_layer // 2)\n",
        "        dirs = generate_biased_directions(y_lp, frac_idx, dirs_count, alpha_k, bias_strength=0.5)\n",
        "\n",
        "        def evaluate_direction(u):\n",
        "            best_loc = local_best\n",
        "            best_loc_y = local_best_y\n",
        "            for sign in [1, -1]:\n",
        "                y_cand = y_frac + sign * alpha_k * u\n",
        "                y_cand_int = (y_cand > 0.5).astype(int)\n",
        "                y_full = y_center.copy()\n",
        "                y_full[frac_idx] = y_cand_int\n",
        "                y_int = np.round(y_full).astype(int)\n",
        "                cost = solve_lp_fixed_y_uflp(y_int, f, c)\n",
        "                if cost < best_loc:\n",
        "                    best_loc = cost\n",
        "                    best_loc_y = y_int.copy()\n",
        "            return best_loc, best_loc_y\n",
        "\n",
        "        with ThreadPoolExecutor(max_workers=NUM_WORKERS) as executor:\n",
        "            futures = [executor.submit(evaluate_direction, dirs[i]) for i in range(dirs_count)]\n",
        "            for future in as_completed(futures):\n",
        "                cost, y_cand = future.result()\n",
        "                if cost < local_best:\n",
        "                    local_best = cost\n",
        "                    local_best_y = y_cand\n",
        "        y_center = local_best_y.copy()\n",
        "        y_frac = y_center[frac_idx].copy()\n",
        "    return local_best, local_best_y\n",
        "\n",
        "def smooth(y, f, c, y_lp=None, iters=1, gap_threshold=5.0):\n",
        "    best = solve_lp_fixed_y_uflp(y, f, c)\n",
        "    best_y = y.copy()\n",
        "    for _ in range(iters):\n",
        "        R_val = compute_R(y)\n",
        "        n_free = len(get_fractional_indices(y))\n",
        "        new_cost, new_y = hierarchical_radial_scan_parallel(\n",
        "            y, R_val, n_free, f, c, best, best_y,\n",
        "            n_layers=2, dirs_per_layer=8, alpha_schedule='adaptive',\n",
        "            y_lp=y_lp, gap_threshold=gap_threshold\n",
        "        )\n",
        "        if new_cost < best:\n",
        "            best = new_cost\n",
        "            best_y = new_y\n",
        "        else:\n",
        "            break\n",
        "    return best, best_y\n",
        "\n",
        "def vcycle(y, f, c, coarse, y_lp=None, gap_threshold=5.0):\n",
        "    cost1, y1 = smooth(y, f, c, y_lp=y_lp, iters=1, gap_threshold=gap_threshold)\n",
        "    if not coarse:\n",
        "        return cost1, y1\n",
        "    best = cost1\n",
        "    best_y = y1\n",
        "    n_coarse = len(coarse)\n",
        "\n",
        "    def evaluate_bits(bits):\n",
        "        yc = np.array([(bits >> i) & 1 for i in range(n_coarse)])\n",
        "        y_full = y1.copy()\n",
        "        for idx, val in zip(coarse, yc):\n",
        "            y_full[idx] = val\n",
        "        cost = solve_lp_fixed_y_uflp(y_full, f, c)\n",
        "        return cost, y_full\n",
        "\n",
        "    with ThreadPoolExecutor(max_workers=NUM_WORKERS) as executor:\n",
        "        futures = [executor.submit(evaluate_bits, bits) for bits in range(1 << n_coarse)]\n",
        "        for future in as_completed(futures):\n",
        "            cost, y_cand = future.result()\n",
        "            if cost < best:\n",
        "                best = cost\n",
        "                best_y = y_cand\n",
        "    cost2, y2 = smooth(best_y, f, c, y_lp=y_lp, iters=1, gap_threshold=gap_threshold)\n",
        "    return cost2, y2\n",
        "\n",
        "def read_koerkel_ghosh_from_text(text):\n",
        "    \"\"\"قراءة ملف Körkel-Ghosh (تجاهل سطور FILE:)\"\"\"\n",
        "    lines = text.strip().splitlines()\n",
        "    lines = [l.strip() for l in lines if l.strip()]\n",
        "    start_idx = -1\n",
        "    n, m = None, None\n",
        "    for idx, line in enumerate(lines):\n",
        "        if line.upper().startswith('FILE:'):\n",
        "            continue\n",
        "        parts = line.split()\n",
        "        if len(parts) >= 2:\n",
        "            try:\n",
        "                n_cand = int(parts[0])\n",
        "                m_cand = int(parts[1])\n",
        "                if n_cand > 0 and m_cand > 0:\n",
        "                    n, m = n_cand, m_cand\n",
        "                    start_idx = idx\n",
        "                    break\n",
        "            except:\n",
        "                continue\n",
        "    if n is None or m is None:\n",
        "        raise ValueError(\"لم يتم العثور على سطر صالح يحتوي على n و m.\")\n",
        "    f = np.zeros(n, dtype=float)\n",
        "    c = np.zeros((n, m), dtype=float)\n",
        "    for i in range(n):\n",
        "        line = lines[start_idx + 1 + i]\n",
        "        parts = line.split()\n",
        "        idx = int(parts[0]) - 1\n",
        "        f[idx] = float(parts[1])\n",
        "        for j in range(m):\n",
        "            c[idx, j] = float(parts[2 + j])\n",
        "    return f, c, n, m\n",
        "\n",
        "def generate_random_instance(n, m):\n",
        "    \"\"\"توليد مسألة UFLP عشوائية\"\"\"\n",
        "    f = np.random.uniform(1000, 20000, n)\n",
        "    c = np.random.uniform(100, 500, (n, m))\n",
        "    return f, c\n",
        "\n",
        "def solve_ahrh(f, c, max_cycles, k_coarse, patience):\n",
        "    \"\"\"الدالة الرئيسية لحل المسألة مع إمكانية تعديل المعاملات\"\"\"\n",
        "    n, m = len(f), c.shape[1]\n",
        "    y_lp, lp_val = lp_relaxation_uflp(f, c)\n",
        "    if lp_val is None:\n",
        "        lp_val = float('inf')\n",
        "    y = np.ones(n, dtype=int)\n",
        "    best = solve_lp_fixed_y_uflp(y, f, c)\n",
        "    gap_history = []\n",
        "    cycles_done = 0\n",
        "    no_improve = 0\n",
        "    start_time = time.time()\n",
        "    for cycle in range(max_cycles):\n",
        "        if y_lp is not None:\n",
        "            open_now = np.where(y > 0.5)[0].tolist()\n",
        "            top_lp = np.argsort(-y_lp)[:k_coarse].tolist()\n",
        "            coarse = list(set(open_now + top_lp))\n",
        "            if len(coarse) > 10:\n",
        "                importance = [(i, y_lp[i]) for i in coarse]\n",
        "                importance.sort(key=lambda x: x[1], reverse=True)\n",
        "                coarse = [i for i, _ in importance[:10]]\n",
        "        else:\n",
        "            coarse = []\n",
        "        new_cost, new_y = vcycle(y, f, c, coarse, y_lp=y_lp, gap_threshold=3.0)\n",
        "        gap = (new_cost - lp_val) / lp_val * 100 if lp_val != float('inf') else 0\n",
        "        gap_history.append(gap)\n",
        "        if new_cost < best - 1e-6:\n",
        "            best = new_cost\n",
        "            y = new_y\n",
        "            no_improve = 0\n",
        "        else:\n",
        "            no_improve += 1\n",
        "            if no_improve >= patience:\n",
        "                cycles_done = cycle + 1\n",
        "                break\n",
        "    if cycles_done == 0:\n",
        "        cycles_done = max_cycles\n",
        "    total_time = time.time() - start_time\n",
        "    return {\n",
        "        'best_cost': best,\n",
        "        'lp_val': lp_val,\n",
        "        'gap': (best - lp_val) / lp_val * 100 if lp_val != float('inf') else 0,\n",
        "        'open_fac': len(np.where(y > 0.5)[0]),\n",
        "        'cycles_done': cycles_done,\n",
        "        'gap_history': gap_history,\n",
        "        'total_time': total_time\n",
        "    }\n",
        "\n",
        "# ===================== واجهة Streamlit =====================\n",
        "st.set_page_config(page_title=\"AHRH Solver\", layout=\"wide\")\n",
        "st.title(\"🧠 AHRH: خوارزمية هرمية انكماشية لحل مسائل UFLP\")\n",
        "st.markdown(\"---\")\n",
        "\n",
        "# إدخال المعاملات\n",
        "with st.sidebar:\n",
        "    st.header(\"⚙️ معاملات الخوارزمية\")\n",
        "    max_cycles = st.slider(\"عدد الدورات الأقصى\", 5, 50, 15, 5)\n",
        "    k_coarse = st.slider(\"حجم المجموعة الخشنة (k)\", 3, 10, 5)\n",
        "    patience = st.slider(\"الصبر (عدد الدورات بدون تحسن)\", 2, 10, 3)\n",
        "    st.markdown(\"---\")\n",
        "    st.write(\"عدد العمال المستخدم في التوازي:\", NUM_WORKERS)\n",
        "\n",
        "col1, col2 = st.columns([1, 2])\n",
        "\n",
        "with col1:\n",
        "    st.header(\"📂 مصدر المسألة\")\n",
        "    option = st.radio(\"اختر:\", (\"توليد عشوائي\", \"رفع ملف\"))\n",
        "\n",
        "    if option == \"توليد عشوائي\":\n",
        "        n = st.number_input(\"عدد المواقع (n)\", min_value=5, max_value=200, value=50, step=5)\n",
        "        m = st.number_input(\"عدد العملاء (m)\", min_value=5, max_value=200, value=50, step=5)\n",
        "        if st.button(\"🚀 توليد وحل\"):\n",
        "            with st.spinner(\"جاري توليد المسألة وتشغيل الخوارزمية...\"):\n",
        "                f, c = generate_random_instance(int(n), int(m))\n",
        "                result = solve_ahrh(f, c, max_cycles, k_coarse, patience)\n",
        "                st.session_state['result'] = result\n",
        "                st.session_state['n'] = n\n",
        "                st.session_state['m'] = m\n",
        "    else:\n",
        "        uploaded_file = st.file_uploader(\"ارفع ملف المسألة (txt)\", type=[\"txt\"])\n",
        "        if uploaded_file is not None:\n",
        "            with st.spinner(\"جاري قراءة الملف وتشغيل الخوارزمية...\"):\n",
        "                text = uploaded_file.getvalue().decode(\"utf-8\")\n",
        "                try:\n",
        "                    f, c, n, m = read_koerkel_ghosh_from_text(text)\n",
        "                    result = solve_ahrh(f, c, max_cycles, k_coarse, patience)\n",
        "                    st.session_state['result'] = result\n",
        "                    st.session_state['n'] = n\n",
        "                    st.session_state['m'] = m\n",
        "                except Exception as e:\n",
        "                    st.error(f\"خطأ في قراءة الملف: {e}\")\n",
        "\n",
        "with col2:\n",
        "    st.header(\"📊 النتائج\")\n",
        "    if 'result' in st.session_state:\n",
        "        res = st.session_state['result']\n",
        "        colA, colB, colC, colD = st.columns(4)\n",
        "        colA.metric(\"أفضل تكلفة\", f\"{res['best_cost']:,.0f}\")\n",
        "        colB.metric(\"قيمة LP\", f\"{res['lp_val']:,.0f}\")\n",
        "        colC.metric(\"الفجوة\", f\"{res['gap']:.4f}%\")\n",
        "        colD.metric(\"المرافق المفتوحة\", res['open_fac'])\n",
        "\n",
        "        colE, colF, colG = st.columns(3)\n",
        "        colE.metric(\"عدد الدورات\", res['cycles_done'])\n",
        "        colF.metric(\"الزمن (ث)\", f\"{res['total_time']:.2f}\")\n",
        "        colG.metric(\"حجم المسألة\", f\"{st.session_state['n']}×{st.session_state['m']}\")\n",
        "\n",
        "        if res['gap_history']:\n",
        "            st.subheader(\"📈 تطور الفجوة خلال الدورات\")\n",
        "            fig, ax = plt.subplots(figsize=(10, 5))\n",
        "            cycles = list(range(1, len(res['gap_history'])+1))\n",
        "            ax.plot(cycles, res['gap_history'], marker='o', linestyle='-', color='b')\n",
        "            ax.set_xlabel(\"الدورة\")\n",
        "            ax.set_ylabel(\"الفجوة (%)\")\n",
        "            ax.set_title(\"تطور الفجوة\")\n",
        "            ax.grid(True)\n",
        "            st.pyplot(fig)\n",
        "\n",
        "            df = pd.DataFrame({\"Cycle\": cycles, \"Gap (%)\": res['gap_history']})\n",
        "            csv = df.to_csv(index=False)\n",
        "            st.download_button(\n",
        "                label=\"📥 تحميل تطور الفجوة (CSV)\",\n",
        "                data=csv,\n",
        "                file_name=\"gap_evolution.csv\",\n",
        "                mime=\"text/csv\"\n",
        "            )\n",
        "    else:\n",
        "        st.info(\"👈 اختر مصدر المسألة واضغط على زر التشغيل لعرض النتائج.\")\n",
        "\n",
